{\begin{enumerate}
\item		$x=\cos t$\quad $y=\sin t$, \quad $0\leq t\leq 2\pi$
\item		$x=\cos (t^2)$\quad $y=\sin(t^2)$, \quad $0\leq t\leq 2\pi$
\item		$x=\cos (1/t)$\quad $y=\sin(1/t)$, \quad $0<t<1$
\item		$x=\cos(\cos t)$\quad $y=\sin(\cos t)$, \quad $0\leq t\leq 2\pi$
\end{enumerate}
