{$\coth^2 x-\csch^2x=1$}
{\hfill$\begin{aligned}[t]
		\coth^2x-\csch^2x &= \left(\frac{e^x+e^{-x}}{e^x-e^{-x}}\right)^2 - \left(\frac{2}{e^x-e^{-x}}\right)^2 \\
											&= \frac{(e^{2x} + 2 + e^{-2x}) - (4)}{e^{2x} - 2 + e^{-2x}}\\
											&= \frac{e^{2x} - 2 + e^{-2x}}{e^{2x} - 2 + e^{-2x}}\\
											&= 1										
\end{aligned}$\hfill\null}
